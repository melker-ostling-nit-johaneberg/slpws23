def Check_admin(current_user)
    db = Connect_to_db("db/db.db")
    a = db.execute("SELECT Admin FROM Users WHERE User_Id = ?", current_user).last
    if a == nil
        return false
    else
        return a["Admin"] == 1
    end
end
